```typescript
// store.tsx linha ~200
supabase
  .channel('public_updates')
  .on('postgres_changes',
    { event: '*', schema: 'public', table: 'admin_banners' },
    (payload) => { fetchBanners(); }
  )
  .subscribe();

// Resultado:
// Admin muda banner вҶ’ Banco atualiza вҶ’ Realtime notifica вҶ’ fetchBanners() вҶ’ state atualiza вҶ’ LawyerDashboard re-renderiza
```
